# Setup Instructions
#
# STEP 0: Bluetooth Hardware Setup
# ===================================================
# sudo rfkill list                    # check if bluetooth is blocked
# sudo rfkill unblock bluetooth       # unblock if needed
# sudo rfkill unblock all
#
# STEP 1: System Packages (run on Raspberry Pi OS)
# ========================================================
# sudo apt-get update && sudo apt-get upgrade -y
# sudo apt-get install -y python3 python3-pip python3-dev
# sudo apt-get install -y git curl wget
# sudo apt-get install -y bluez bluez-tools
# sudo apt-get install -y python3-bluez python3-dbus libbluetooth-dev
# sudo apt-get install -y aircrack-ng mdk4 wifite
# sudo apt-get install -y dnsmasq hostapd iptables
# sudo apt install python3-bleak
#
# STEP 2: Initialize Bluetooth Adapter
# =================================================
# sudo hciconfig hci0 reset           # reset adapter
# hciconfig hci0                      # verify it came back up
# sudo btmgmt power on                # power on adapter
# sudo btmgmt connectable on          # enable connectable mode
# sudo btmgmt discov on               # enable discoverable mode
# sudo btmgmt pairable on             # enable pairing mode
#
# STEP 3: WiFi Adapter Setup (TP-Link Archer T2U PLUS RTL8821AU)
# ================================================================
# sudo apt-get update
# sudo apt-get install -y git build-essential dkms bc linux-headers-generic
#
# cd ~
# git clone https://github.com/morrownr/8821au-20210708.git
# cd 8821au-20210708
#
# sudo ./install-driver.sh
# When prompted "Do you want to edit the driver options file now?" → Type n
#
# sudo reboot
#

# iwconfig
